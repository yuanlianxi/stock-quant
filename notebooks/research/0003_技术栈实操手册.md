akshare 是目前最流行的免费财经数据接口，支持股票、期货、期权、基金等多品种数据。

```python

print(df_daily.head())
# 字段：日期、开盘、收盘、最高、最低、成交量、成交额、振幅、涨跌幅等
```

## 2. Backtrader 回测框架

### 2.1 基础模板

```python
import backtrader as bt
import pandas as pd
import akshare as ak

# Step 1: 获取数据
df = ak.stock_zh_a_hist(symbol="000001", period="daily",
                         start_date="20230101", end_date="20250601", adjust="qfq")

# 数据格式转换（Backtrader 要求列名小写+标准化）
df.columns = [c.lower() for c in df.columns]
df = df.rename(columns={
    'date': 'datetime',
    'code': 'code'
})
df['datetime'] = pd.to_datetime(df['datetime'])
df.set_index('datetime', inplace=True)
df = df[['open', 'high', 'low', 'close', 'volume']]

# Step 2: 定义策略
class MyStrategy(bt.Strategy):
    params = (
        ('ma_period', 20),
    )

    def __init__(self):
        self.ma = bt.indicators.SimpleMovingAverage(
            self.data.close, period=self.params.ma_period
        )

    def next(self):
        if self.data.close[0] > self.ma[0] and not self.position:
            self.buy()
        elif self.data.close[0] < self.ma[0] and self.position:
            self.close()

# Step 3: 配置回测引擎
cerebro = bt.Cerebro()
cerebro.addstrategy(MyStrategy, ma_period=20)

# 添加数据
data = bt.feeds.PandasData(dataname=df)
cerebro.adddata(data)

# 设置初始资金
cerebro.broker.setcash(100000.0)

# 运行回测
cerebro.run()
print(f'Final Portfolio Value: {cerebro.broker.getvalue():.2f}')
```

### 2.2 完整回测模板（双均线 + 布林带止损）

```python
import backtrader as bt
import pandas as pd
import akshare as ak


class DualMAStopStrategy(bt.Strategy):
    """双均线策略 + 布林带止损"""

    params = (
        ('fast', 5),          # 快速均线周期
        ('slow', 20),         # 慢速均线周期
        ('boll_period', 20),  # 布林带周期
        ('boll_dev', 2),      # 布林带标准差倍数
        ('stop_loss', 0.05),  # 止损比例 5%
    )

    def __init__(self):
        self.ma_fast = bt.indicators.SMA(self.data.close, period=self.params.fast)
        self.ma_slow = bt.indicators.SMA(self.data.close, period=self.params.slow)
        self.boll = bt.indicators.BollingerBands(
            self.data.close,
            period=self.params.boll_period,
            devfactor=self.params.boll_dev
        )
        self.crossover = bt.indicators.CrossOver(self.ma_fast, self.ma_slow)

        # 订单跟踪
        self.order = None

    def log(self, txt, dt=None):
        dt = dt or self.datas[0].datetime.date(0)
        print(f'{dt.isoformat()} {txt}')

    def notify_order(self, order):
        if order.status in [order.Submitted, order.Accepted]:
            return
        if order.status in [order.Completed]:
            if order.isbuy():
                self.log(f'BUY EXECUTED, Price: {order.executed.price:.2f}')
            elif order.issell():
                self.log(f'SELL EXECUTED, Price: {order.executed.price:.2f}')
        elif order.status in [order.Canceled, order.Margin, order.Rejected]:
            self.log('Order Canceled/Margin/Rejected')
        self.order = None

    def next(self):
        if self.order:
            return

        if not self.position:
            # 金叉买入
            if self.crossover > 0:
                self.order = self.buy()
        else:
            # 止损条件：跌破布林下轨 或 价格低于成本价*(1-止损比例)
            stop_price = self.position.price * (1 - self.params.stop_loss)
            if self.data.close[0] < self.boll.lines.bot[0] or \
               self.data.close[0] < stop_price:
                self.order = self.close()


def prepare_data(symbol, start, end):
    """获取并预处理数据"""
    df = ak.stock_zh_a_hist(symbol=symbol, period="daily",
                            start_date=start, end_date=end, adjust="qfq")
    df.columns = [c.lower() for c in df.columns]
    df['datetime'] = pd.to_datetime(df['date'])
    df.set_index('datetime', inplace=True)
    return df[['open', 'high', 'low', 'close', 'volume']]


if __name__ == '__main__':
    cerebro = bt.Cerebro()

    # 添加策略
    cerebro.addstrategy(DualMAStopStrategy)

    # 添加数据
    df = prepare_data("000001", "20230101", "20250601")
    data = bt.feeds.PandasData(dataname=df)
    cerebro.adddata(data)

    # 初始资金和佣金
    cerebro.broker.setcash(100000.0)
    cerebro.broker.setcommission(commission=0.001)

    # 打印初始资金
    print(f'Starting Portfolio Value: {cerebro.broker.getvalue():.2f}')
    cerebro.run()
    print(f'Final Portfolio Value: {cerebro.broker.getvalue():.2f}')

    # 绘图
    cerebro.plot()
```

## 3. 数据存储

### 3.1 SQLite 存储（开发/回测用）

```python
import sqlite3
import pandas as pd

DB_PATH = 'data/stock_data.db'


def get_conn():
    return sqlite3.connect(DB_PATH)


def save_kline_daily(df, table_name='kline_daily'):
    """存储日线数据"""
    conn = get_conn()
    df.to_sql(table_name, conn, if_exists='append', index=False)
    conn.close()


def load_kline_daily(code, table_name='kline_daily'):
    """读取指定股票日线数据"""
    conn = get_conn()
    df = pd.read_sql(
        'SELECT * FROM kline_daily WHERE code=? ORDER BY date ASC',
        conn,
        params=(code,)
    )
    conn.close()
    return df


def save_batch(df_list, table_name='kline_daily'):
    """批量存储（合并去重）"""
    conn = get_conn()
    for df in df_list:
        # 先查已存在的日期
        existing = pd.read_sql(
            f'SELECT DISTINCT date FROM {table_name}',
            conn
        )['date'].tolist()
        df_new = df[~df['date'].isin(existing)]
        if not df_new.empty:
            df_new.to_sql(table_name, conn, if_exists='append', index=False)
    conn.close()
```

### 3.2 PostgreSQL 存储（生产环境）

```python
from sqlalchemy import create_engine, text
import pandas as pd

# 生产数据库连接
ENGINE = create_engine(
    'postgresql://user:password@localhost:5432/stock_db',
    pool_size=10,
    max_overflow=20
)


def save_kline_pg(df, table_name='kline_daily'):
    """存储到 PostgreSQL"""
    df.to_sql(
        table_name,
        ENGINE,
        if_exists='append',
        index=False,
        method='multi',
        chunksize=1000
    )


def load_kline_pg(code, start_date, end_date, table_name='kline_daily'):
    """从 PostgreSQL 读取"""
    query = text('''
        SELECT * FROM kline_daily
        WHERE code = :code AND date BETWEEN :start AND :end
        ORDER BY date
    ''')
    with ENGINE.connect() as conn:
        df = pd.read_sql(query, conn, params={
            'code': code,
            'start': start_date,
            'end': end_date
        })
    return df


def batch_update_status(code, dates, status, table_name='kline_daily'):
    """批量更新处理状态"""
    with ENGINE.connect() as conn:
        for date in dates:
            conn.execute(
                text(f'''
                    UPDATE {table_name}
                    SET fetch_status = :status, fetch_at = NOW()
                    WHERE code = :code AND date = :date
                '''),
                {'status': status, 'code': code, 'date': date}
            )
        conn.commit()
```

### 3.3 数据表结构设计

```sql
-- 日线数据表
CREATE TABLE IF NOT EXISTS kline_daily (
    id          SERIAL PRIMARY KEY,
    code        VARCHAR(10) NOT NULL,          -- 股票代码 000001
    name        VARCHAR(50),                    -- 股票名称
    date        DATE NOT NULL,                  -- 交易日期
    open        DECIMAL(10, 3),                 -- 开盘价
    high        DECIMAL(10, 3),                 -- 最高价
    low         DECIMAL(10, 3),                 -- 最低价
    close       DECIMAL(10, 3),                 -- 收盘价
    volume      BIGINT,                         -- 成交量
    amount      DECIMAL(20, 2),                 -- 成交额
    adjust_flag VARCHAR(5),                     -- 复权类型 qfq/hfq
    created_at  TIMESTAMP DEFAULT NOW(),

    -- 处理状态（用于增量更新）
    fetch_status TEXT DEFAULT 'pending',       -- pending/processing/completed/failed
    fetch_at     TIMESTAMP,
    fetch_error  TEXT,

    UNIQUE(code, date, adjust_flag)
);

CREATE INDEX idx_kline_code_date ON kline_daily(code, date);
CREATE INDEX idx_kline_fetch_status ON kline_daily(fetch_status);
```

## 常见错误速查

| 错误信息 | 原因 | 解决方案 |
|---------|------|---------|
| `akshare: 403 Forbidden` | 请求频率过高 | 添加 `time.sleep(1)` 间隔 |
| `baostock: login() failed` | 服务器维护/网络问题 | 稍后重试 |
| `tushare: invalid token` | Token 错误或过期 | 重新设置 token |
| `backtrader: No data returned` | 数据为空 | 检查股票代码和日期格式 |
| `sqlite3: database is locked` | 并发写入冲突 | 使用连接池或串行写入 |
| `ModuleNotFoundError` | 依赖未安装 | `pip install xxx` |
| `Date parse error` | 日期格式不一致 | 使用 `pd.to_datetime()` 统一转换 |
| `position.size not available` | 访问时机过早 | 在 `next()` 中使用，不要在 `__init__` 中使用 |
