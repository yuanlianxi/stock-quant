## 四、因子计算层

### 4.1 ATR计算 (atr.py)

**输入**：日线 OHLCV
**输出**：N值序列

```python
def calculate_atr(df: pd.DataFrame, period: int = 20) -> pd.Series:
    """
    TR = MAX(H-L, |H-PDC|, |L-PDC|)
    N[1] = SMA(TR, 20)
    N[t] = (N[t-1] × 19 + TR[t]) / 20
    """
    df = df.copy()
    df['TR'] = df.apply(
        lambda row: max(
            row['high'] - row['low'],
            abs(row['high'] - row['close'].shift(1)),
            abs(row['low'] - row['close'].shift(1))
        ), axis=1
    )
    df['N'] = df['TR'].ewm(alpha=1/period, adjust=False).mean()
    return df['N']
```

### 4.2 突破信号 (breakout.py)

```python
def calculate_breakout(df: pd.DataFrame, lookback: int = 55) -> pd.Series:
    """
    55日最高价突破 → 多头信号 (+1)
    55日最低价下穿 → 空头信号 (-1)
    否则 → 0
    """
    df = df.copy()
    df['highest_55'] = df['high'].rolling(lookback).max()
    df['lowest_55'] = df['low'].rolling(lookback).min()
    
    conditions = [
        df['close'] > df['highest_55'].shift(1),
        df['close'] < df['lowest_55'].shift(1)
    ]
    choices = [1, -1]
    return pd.Series(np.select(conditions, choices, default=0), index=df.index)
```

### 4.3 离市信号 (exit.py)

```python
def calculate_exit_signal(df: pd.DataFrame, lookback: int = 20) -> pd.Series:
    """
    多头：20日最低价被下穿 → 退出 (+2，标记为多头止盈/止损)
    空头：20日最高价被上穿 → 退出 (-2，标记为空头止盈/止损)
    否则 → 0
    """
    df = df.copy()
    df['lowest_20'] = df['low'].rolling(lookback).min()
    df['highest_20'] = df['high'].rolling(lookback).max()
    
    conditions = [
        df['close'] < df['lowest_20'].shift(1),  # 多头退出
        df['close'] > df['highest_20'].shift(1)   # 空头退出
    ]
    choices = [2, -2]
    return pd.Series(np.select(conditions, choices, default=0), index=df.index)
```

## 五、仓位管理层

### 5.1 头寸单位计算 (size_calculator.py)

```python
def calc_position_unit(total_equity: float, n: float, point_value: float, risk_pct: float = 0.01) -> float:
    """
    单位 = (总资金 × 1%) / (N × 每点人民币)
    
    参数：
    - total_equity: 总权益
    - n: ATR值
    - point_value: 每点价值（元）
    - risk_pct: 单笔风险比例，默认1%
    
    示例：总资金100万，N=1.20原油（每点10元）
    → 单位 = 1,000,000 × 1% / (1.20 × 10) = 833手
    """
    return (total_equity * risk_pct) / (n * point_value)
```

### 5.2 海龟仓位管理器 (turtle_position.py)

```python
class TurtlePositionManager:
    MAX_UNITS = 4          # 最大4个单位
    ADD_INTERVAL = 0.5     # 加仓间隔：0.5N
    
    def __init__(self, max_units: int = 4):
        self.max_units = max_units
        self.positions: Dict[str, Position] = {}
    
    def on_breakout_entry(self, symbol: str, direction: int, price: float, n: float, total_equity: float, point_value: float):
        """55日突破时调用，建立第一个单位"""
        unit = calc_position_unit(total_equity, n, point_value)
        stop_loss = price - 2 * n * direction  # 2N止损
        
        self.positions[symbol] = Position(
            symbol=symbol, direction=direction,
            units=1, entry_price=price,
            current_unit_entry=[price],
            stop_loss=stop_loss, atr=n
        )
    
    def on_add_unit(self, symbol: str, price: float, n: float):
        """符合加仓条件时调用（第2-4个单位）"""
        pos = self.positions.get(symbol)
        if not pos or pos.units >= self.max_units:
            return
        
        expected_entry = pos.current_unit_entry[-1] + self.ADD_INTERVAL * n * pos.direction
        # 若跳空导致无法在设定点位成交，以市场价直接增加
        pos.units += 1
        pos.current_unit_entry.append(price)
        
        # 动态调整所有单位止损：上移1/2N（跳空特殊处理不做调整）
        self._update_stop_loss(pos, n, skip_adjustment=False)
    
    def on_exit(self, symbol: str, exit_price: float):
        """20日反穿时，平全部单位"""
        pos = self.positions.pop(symbol, None)
        return pos
    
    def _update_stop_loss(self, pos: Position, n: float, skip_adjustment: bool = False):
        """动态调整止损：每加一单位，前序仓位止损上移1/2N"""
        if skip_adjustment:
            return
        # 基础止损 + (units-1) × 0.5N
        base = pos.current_unit_entry[0] - 2 * n * pos.direction
        new_stop = base - pos.direction * (pos.units - 1) * 0.5 * n
        pos.stop_loss = new_stop
```

### 6.1 信号生成 (signals.py)

```python
class SignalGenerator:
    def __init__(self, lookback_entry: int = 55, lookback_exit: int = 20):
        self.lookback_entry = lookback_entry
        self.lookback_exit = lookback_exit
    
    def generate(self, df: pd.DataFrame) -> pd.DataFrame:
        df['atr'] = calculate_atr(df)
        df['entry_signal'] = calculate_breakout(df, self.lookback_entry)
        df['exit_signal'] = calculate_exit_signal(df, self.lookback_exit)
        
        # 合并：entry_signal ≠ 0 → 产生入市信号
        #        exit_signal ≠ 0 → 产生离市信号
        return df
```

### 6.2 订单生成 (order_generator.py)

```python
def generate_orders(symbol: str, signal: int, price: float, n: float, quantity: float) -> Order:
    """
    signal:
      +1: 55日突破 → 开多
      -1: 55日跌破 → 开空
      +2: 20日多头反穿 → 平多
      -2: 20日空头反穿 → 平空
    """
    if signal == 1:
        return Order(symbol, SIDE.BUY, price, quantity, ORDER_TYPE.MARKET)
    elif signal == -1:
        return Order(symbol, SIDE.SELL, price, quantity, ORDER_TYPE.MARKET)
    elif signal == 2:
        return Order(symbol, SIDE.SELL, price, quantity, ORDER_TYPE.MARKET)
    elif signal == -2:
        return Order(symbol, SIDE.BUY, price, quantity, ORDER_TYPE.MARKET)
```

## 七、风控层

### 7.1 单笔风控 (single_trade_risk.py)

```python
def check_single_trade_risk(position: Position, current_price: float, n: float, max_risk_pct: float = 0.02) -> bool:
    """
    单笔最大风险2%
    止损幅度 = 2N
    若当前价格触发止损 → 触发平仓
    """
    loss_per_unit = abs(current_price - position.entry_price)
    risk_pct = loss_per_unit * position.point_value * position.units / position.total_equity
    return risk_pct > max_risk_pct
```

### 7.2 账户风控 (account_risk.py)

```python
RISK_RULES = {
    'max_risk_exposure': 0.90,      # 风险度>90%强制减仓
    'resume_risk_threshold': 0.80, # 砍仓后风险度<80%才能重新开仓
    'force_liquidation': 0.70,     # 净值<70%强制清盘
}
```

```python
def apply_account_risk(equity: float, initial_equity: float, risk_exposure: float, positions: List[Position]) -> RiskAction:
    nav = equity / initial_equity
    
    # 强制清盘
    if nav < RISK_RULES['force_liquidation']:
        return RiskAction.FORCE_LIQUIDATE
    
    # 风险度超标
    if risk_exposure > RISK_RULES['max_risk_exposure']:
        return RiskAction.CUT_POSITION
    
    # 重新开仓检查
    if risk_exposure < RISK_RULES['resume_risk_threshold']:
        return RiskAction.RESUME_ALLOWED
    
    return RiskAction.NORMAL
```

### 7.3 净值管理 (nav_manager.py)

```python
def calc_intrinsic_value(positions: List[Position], cash: float) -> float:
    """
    I.V = max(止损价, 近20日最低价) × 期货持仓 + 现金
    用于判断名义资金是否上调
    """
    iv = cash
    for pos in positions:
        pos_value = pos.current_price * pos.units * pos.point_value
        iv += pos_value
    return iv

def get_nominal_capital(equity: float, initial_equity: float, iv: float, current_nominal: float) -> float:
    """根据I.V和净值确定名义资金"""
    nav = equity / initial_equity
    
    if iv > current_nominal:
        return iv  # 上调
    if nav < 0.90:
        return initial_equity * 0.8
    if nav < 0.80:
        return initial_equity * 0.64
    return current_nominal
```

### 8.2 绩效分析 (performance_analytics.py)
